```jsX
//项目的根组件
//被引入到index.js中，然后渲染到public的index.html(root)中

// import 语法
import { useState } from 'react'
//表单受控绑定
function App() {
  //1.声明一个react状态-useState

  //2.核心绑定流程
  //2.1通过value属性绑定react状态
  //2.2绑定onChange事件 通过事件参数e拿到输入框最新的值，反向修改到react中
  const [value, setValue] = useState('')
  return (
    <div className="App">
      <input value={value}
        onChange={(e) => setValue(e.target.value)}
        type='text' />
    </div >
  );
}

export default App;
```
